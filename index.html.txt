  (function() {
    // ---------- ДАНІ КАТЕГОРІЙ ----------

      1: {
        title: '⛰️ Гори',
        badge: 'категорія 1',
        html: `
          <h3><i>⛰️</i> Висотний напрям</h3>
          <p>Стратегія розвитку компанії в гірському сегменті: інновації, стійкість та амбітні проєкти. Ми підкорюємо вершини разом із нашими партнерами.</p>
          <div class="highlight"><span>🏔️ 3 піки · 2026</span></div>
          <div class="grid-cards">
            <div class="card"><i>🧗</i> Альпінізм</div>
            <div class="card"><i>⛺</i> Табори</div>
            <div class="card"><i>📈</i> Зростання</div>
          </div>
        `
      },
      2: {
        title: '🌄 Передгір\'я',
        badge: 'категорія 2',
        html: `
          <h3><i>🌄</i> Перехідна зона</h3>
          <p>Передгір'я — місце, де народжуються нові рішення. Ми поєднуємо гнучкість рівнин із потенціалом висот, створюючи унікальні сервіси.</p>
          <div class="highlight"><span>🌿 5 напрямків · адаптивність</span></div>
          <div class="grid-cards">
            <div class="card"><i>🧭</i> Навігація</div>
            <div class="card"><i>⚙️</i> Технології</div>
            <div class="card"><i>🤝</i> Партнерство</div>
          </div>
        `
      },
      3: {
        title: '🏞️ Долина',
        badge: 'категорія 3',
        html: `
          <h3><i>🏞️</i> Долина можливостей</h3>
          <p>У долині ми впроваджуємо сталі рішення, розвиваємо громади та будуємо майбутнє. Тут народжується справжня цінність для клієнтів.</p>
          <div class="highlight"><span>🌾 12 проєктів · сталість</span></div>
          <div class="grid-cards">
            <div class="card"><i>🌱</i> Екологія</div>
            <div class="card"><i>🏡</i> Громади</div>
            <div class="card"><i>💡</i> Інновації</div>
          </div>
        `
      }
    };

    // ---------- DOM-посилання ----------
    const categoryBtns = document.querySelectorAll('.category-btn');
    const titleEl = document.getElementById('categoryTitle');
    const badgeEl = document.getElementById('categoryBadge');
    const contentDisplay = document.getElementById('contentDisplay');

    // ---------- Функція перемикання ----------
    function switchCategory(categoryId) {
      const data = contentMap[categoryId];
      if (!data) return;

      // Оновлення заголовка та бейджа
      titleEl.textContent = data.title;
      badgeEl.textContent = data.badge;

      // Оновлення HTML-контенту
      contentDisplay.innerHTML = data.html;

      // Оновлення активного класу на кнопках
      categoryBtns.forEach(btn => {
        const btnCat = btn.getAttribute('data-category');
        if (btnCat === categoryId.toString()) {
          btn.classList.add('active');
        } else {
          btn.classList.remove('active');
        }
      });
    }

    // ---------- Підписка на кліки ----------
    categoryBtns.forEach(btn => {
      btn.addEventListener('click', function(e) {
        const cat = this.getAttribute('data-category');
        if (cat) switchCategory(cat);
      });
    });

    // ---------- Ініціалізація: активна за замовчуванням ----------
    const activeBtn = document.querySelector('.category-btn.active');
    if (activeBtn) {
      const defaultCat = activeBtn.getAttribute('data-category');
      if (defaultCat && contentMap[defaultCat]) {
        switchCategory(defaultCat);
      } else {
        // якщо активна, але немає даних — беремо першу
        const first = document.querySelector('.category-btn');
        if (first) {
          first.classList.add('active');
          const firstCat = first.getAttribute('data-category');
          if (firstCat) switchCategory(firstCat);
        }
      }
    } else {
      // якщо жодна не активна — активуємо першу
      const first = document.querySelector('.category-btn');
      if (first) {
        first.classList.add('active');
        const firstCat = first.getAttribute('data-category');
        if (firstCat) switchCategory(firstCat);
      }
    }
  })();
</script>
